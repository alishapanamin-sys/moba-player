<script>
  // ✅ ضع رابط الـ CPA هنا:
  const CPA_LINK = "https://adsmain.bountyads.store/?utm_medium=45cb5b04ee2dd3d524876ef99d99f52ba4436745&utm_campaign=newphl";

  // ---- Quiz logic
  const opts = document.querySelectorAll(".opt");
  const continueBtn = document.getElementById("continueBtn");
  let answers = {
    active: null,
    region: null
  };

  function updateContinueState(){
    // نسمح بالمتابعة فقط لو: active != "no" و region == "ph_yes"
    const okActive = answers.active && answers.active !== "no";
    const okRegion = answers.region === "ph_yes";
    continueBtn.disabled = !(okActive && okRegion);
  }

  opts.forEach(btn=>{
    btn.addEventListener("click", ()=>{
      // group detection by question block
      const text = btn.textContent.toLowerCase();
      const ans = btn.dataset.answer;

      // Determine which group
      if(ans === "yes" || ans === "sometimes" || ans === "no"){
        answers.active = ans;
        // Clear selected in same group
        document.querySelectorAll('.opt[data-answer="yes"],.opt[data-answer="sometimes"],.opt[data-answer="no"]')
          .forEach(x=>x.classList.remove("selected"));
      }
      if(ans === "ph_yes" || ans === "ph_no"){
        answers.region = ans;
        document.querySelectorAll('.opt[data-answer="ph_yes"],.opt[data-answer="ph_no"]')
          .forEach(x=>x.classList.remove("selected"));
      }

      btn.classList.add("selected");
      updateContinueState();
    });
  });

  continueBtn.addEventListener("click", ()=>{
    if(CPA_LINK.includes("YOUR_CPA_LINK_HERE")){
      alert("Please replace YOUR_CPA_LINK_HERE with your real CPA link.");
      return;
    }
    // Redirect
    window.location.href = CPA_LINK;
  });

  // ---- Countdown timer + slots (simple UX boost)
  let seconds = 10 * 60 - 1; // 09:59
  const timerEl = document.getElementById("timer");
  const slotsEl = document.getElementById("slots");
  let slots = 17;

  function tick(){
    const mm = String(Math.floor(seconds / 60)).padStart(2,"0");
    const ss = String(seconds % 60).padStart(2,"0");
    timerEl.textContent = `${mm}:${ss}`;

    // Every ~75 seconds reduce slots (visual urgency)
    if(seconds % 75 === 0 && seconds !== 0 && slots > 5){
      slots -= 1;
      slotsEl.textContent = slots;
    }

    seconds--;
    if(seconds < 0){
      seconds = 10 * 60 - 1;
      slots = 17;
      slotsEl.textContent = slots;
    }
  }
  setInterval(tick, 1000);
</script>
